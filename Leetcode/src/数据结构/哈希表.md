# 两数之和

[两数之和](https://leetcode-cn.com/problems/two-sum/)

```
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> dict;
        vector<int>  res;
        for(int i=0;i<nums.size();++i)
        {
            if(dict.find(target-nums[i]) != dict.end())
                return {dict[target-nums[i]],i};
            dict[nums[i]] = i;
        }
        return res;
    }
};
```

# 最长连续序列

[最长连续序列](https://leetcode-cn.com/problems/longest-consecutive-sequence/)

```
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        int res  = 0;
        unordered_set<int> st(nums.begin(),nums.end());
        for(auto num: nums)
        {
            if(st.count(num-1))
                continue;   //@ 避免重复，只有前一个数字没有出现时，当前的数字可以作为起始数字
            int cnt = 1;
            while(st.count(++num))
                cnt++;
            res = max(res,cnt);
        }
        return res;
    }
};
```

# 最长和谐子序列

[最长和谐子序列](https://leetcode-cn.com/problems/longest-harmonious-subsequence/)

```
class Solution {
public:
    int findLHS(vector<int>& nums) 
    {
        unordered_map<int,int> dict;
        for(const auto num : nums)
            dict[num]++;
        
        int res  = 0;
        for(auto &it : dict)
        {
            if(dict.count(it.first+1))
                res = max(res,dict[it.first] + dict[it.first + 1]); 
        }
        return res;
    }
};
```

# 重新安排行程

[重新安排行程](https://leetcode-cn.com/problems/reconstruct-itinerary/)

```
class Solution {
public:
    vector<string> findItinerary(vector<vector<string>>& tickets) {
        vector<string> res;
        if(tickets.empty())
            return res;
        
        //@ 同一起点出发，可以到达不同的终点
        unordered_map<string,multiset<string>> hash;
        for(const auto & ticket : tickets)
            hash[ticket[0]].insert(ticket[1]);
        
        stack<string> s;
        s.push("JFK");
        while(!s.empty())
        {
            string next = s.top();
            if(hash[next].empty())
            {
                res.push_back(next);
                s.pop();
            }
            else
            {
                s.push(*hash[next].begin());
                hash[next].erase(hash[next].begin());
            }
        }
        reverse(res.begin(),res.end());
        return res;
    }
};
```

# 146. LRU缓存机制

[146. LRU缓存机制](https://leetcode-cn.com/problems/lru-cache/)

```
class LRUCache {
	list<int> lst_;
	unordered_map<int, int> dict_;
	int cap_;

public:
	LRUCache(int capacity) {
		cap_ = capacity;
	}

	int get(int key) {
		if (dict_.count(key) == 0)
			return -1;

		int res = dict_[key];
		lst_.remove(key);
		lst_.push_front(key);
		return res;
	}

	void put(int key, int value) {
		if (dict_.count(key) == 0)
		{
			if (lst_.size() == cap_)
			{
				dict_.erase(lst_.back());
				lst_.pop_back();
			}
			lst_.push_front(key);
			dict_[key] = value;
		}
		else
		{
			lst_.remove(key);
			lst_.push_front(key);
			dict_[key] = value;
		}
	}
};
```

较快的方式：

```
class LRUCache {
    list<pair<int, int>> lst_;
    unordered_map<int, list<pair<int, int>>::iterator> dict_;
    int cap_;
public:
    LRUCache(int capacity) {
        cap_ = capacity;
    }
    
    int get(int key) {
		if(dict_.count(key) == 0)
			return -1;
		int res = (*dict_[key]).second;
		lst_.erase(dict_[key]);
		lst_.push_front(make_pair(key,res));
		dict_[key] = lst_.begin();
		return res;
    }
    
    void put(int key, int value) {
		if(dict_.count(key) == 0)
		{
			if(lst_.size() == cap_)
			{
				dict_.erase(lst_.back().first);
				lst_.pop_back();
			}
			lst_.push_front(make_pair(key,value));
			dict_[key] = lst_.begin();
		}
		else
		{
			lst_.erase(dict_[key]);
			lst_.push_front(make_pair(key,value));
			dict_[key] = lst_.begin();
		}		
    }
};
```
