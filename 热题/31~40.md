# 合并有序链表

[33 合并有序链表](https://www.nowcoder.com/practice/a479a3f0c4554867b35356e0d57cf03d?tpId=190&&tqId=35188&rp=1&ru=/ta/job-code-high-rd&qru=/ta/job-code-high-rd/question-ranking)

```
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
       if(l1 == nullptr)
		   return l2;
	   if(l2 == nullptr)
		   return l1;
	   
	   ListNode* res = l1->val < l2->val ? l1 : l2;
	   ListNode* curr = res;
	   if(curr == l1)
		   l1 = l1->next;
	   else
		   l2 = l2->next;
	   
	   while(l1 && l2)
	   {
		   if(l1->val < l2->val)
		   {
			   curr->next = l1;
			   l1 = l1->next;
		   }
		   else
		   {
			   curr->next = l2;
			   l2 = l2->next;
		   }
		   curr = curr->next; 		   
	   }
	   
	   if(l1)
		   curr->next = l1;
	   if(l2)
		   curr->next = l2;
	   
	   return res;
    }
};
```
